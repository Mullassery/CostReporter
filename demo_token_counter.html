💻 Usage Example

from pytokencalc.tokenizers import TokenCounterRegistry registry = TokenCounterRegistry() # Count tokens for any model result = registry.count_tokens("gpt-4o", "Your prompt") print(f"Tokens: {result.input_tokens}") print(f"Latency: {result.latency_ms}ms") print(f"Cached: {result.cached}") # Same code for all providers registry.count_tokens("claude-3-5-sonnet", text) registry.count_tokens("gemini-2-flash", text) registry.count_tokens("llama-70b", text)
